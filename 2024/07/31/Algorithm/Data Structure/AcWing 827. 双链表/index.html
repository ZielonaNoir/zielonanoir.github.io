# 双链表

#include <iostream>
#include <algorithm>

using namespace std;

const int N = 1e6 + 10;

int e[N], l[N], r[N], idx;
int m;

void init(){
	r[0] = 1, l[1] = 0;
	idx = 2;
	//idx是当前用到的哪儿个数组下标
}

void add(int k, int x){
	e[idx] = x;
	r[idx] = r[k];
	l[idx] = k;
	l[r[k]] = idx;
	r[k] = idx;
	idx ++;
}
void remove(int k){
	r[l[k]] = r[k];
	l[r[k]] = l[k];
}

int main(){
	init();
	int m;
	cin >> m;
	while(m--){
		string op;
		int k, x;
		cin >> op;
		if(op == "L") {
			cin >> x;
			add(0, x);
		}
		else if(op == "R"){
			cin >> x;
			add(l[1], x);
		}
		else if(op == "D"){
			cin >> k;
			remove(k + 1);
		}
		else if(op == "IL"){
			cin >> k >> x;
			add(l[k + 1], x);
		}
		else{
			cin >> k >> x;
			add(k + 1, x);
		}
	}
	for(int i = r[0]; i != 1; i = r[i]) cout << e[i] << ' ';
	// i = r[0]; 因为左端点是不输出的 要从左端点的下一个开始输出 所以要从 r[0] 开始算
	cout << endl;
	return 0;
}

#include <iostream>
#include <algorithm>

using namespace std;

const int N = 1e6 + 10;

int e[N], l[N], r[N], idx;
int m;

void init(){
    r[0] = 1, l[1] = 0;
    //因为 0 和 1 已经被用过所以 idx 用 2
    idx = 2;
}

void add(int k, int x){
    e[idx] = x;
    r[idx] = r[k];
    l[idx] = k;
    l[r[k]] = idx;
    r[k] = idx;
    idx ++;
}

void remove(int k){
    r[l[k]] = r[k];
    l[r[k]] = l[k];
}

int main(){
    cin >> m;
    
    init();
    
    while(m--){
        string op;
        int k, x;
        cin >> op;
        
        if(op == "L"){
            cin >> x;
            add(0, x);
        }
        else if(op == "R"){
            cin >> x;
            add(l[1], x);
        }
        else if(op == "D"){
            cin >> k;
            remove(k + 1);
        }
        else if(op == "IL"){
            cin >> k >> x;
            add(l[k + 1], x);
        }
        else{
            cin >> k >> x;
            add(k + 1, x);
        }
    }
    for(int i = r[0]; i != 1; i = r[i]){
        printf("%d ", e[i]);
    }
    printf("\n");
    return 0;
}
