#include <cstdio>
#include <iostream>

using namespace std;

int main(){
    int n;
    cin >> n;
    
    int c = 0, r = 0, f = 0;
    for(int i = 0; i < n; i++){ //这里的 n 是第一个n 在for外面的 一共多少的数量
        int k;
        char t;
        // scanf("%d %c", &k, &t); //scanf 读入的时候是不会过滤空格、回车、和制表符 也就是tab键的 
                                //所以需要手动给他过滤一下
        cin >> k >> t;
        if(t == 'C') c += k;
        else if (t == 'R') r += k;
        else f += k;
    }
    int sum = c + r + f;
    printf("Total: %d animals\n", sum);
    printf("Total coneys: %d\n", c);
    printf("Total rats: %d\n", r);
    printf("Total frogs: %d\n", f);
    printf("Percentage of coneys: %.2lf %%\n", (double)c / sum * 100 );//y总小知识 %转义比较特殊 %% 
                                            //为了转换成真实的浮点数的除法 我们需要将一个变量转换成double 
                                            //或者是两个都无所谓
    printf("Percentage of rats: %.2lf %%\n", 1.0 * r / sum * 100 );//强制类型转换 隐式转换 乘个浮点数
    printf("Percentage of frogs: %.2lf %%\n", (double)f / sum * 100 ); //y总小知识 在符号相同的情况下
                                                                        //可能从左到右算也可能从右到左算
                                                                        // s = (double)c + r + f; 这是有区别的
}

// A = 2, B = 3 使用scanf实现 scanf("A = %d, B = %d", &a, &b);
