#include <cstdio>

using namespace std;

int main(){
    char s[100];
    string m;
    
    // scanf("%s", s);//这里不要加取地址符号 因为这个s本身就是一个地址
    //cin >> s + 1; 或者也可以用cin //s+1指想读到从下标1开始
    //会读到空格或者回车为止
    
    
    // gets() 这个函数被淘汰掉了 因为不安全
    fgets(s, 10000, stdin); //第一个参数字符数组 第二个参数最多读多少 第三个参数固定stdin 后面讲文件会讲什么东西  可以用fgets读一整行
    
    // cin.getline(s, 10000);这个东西也是行的 第二个参数是最多搞多少
    
    //用getline得包含string头文件 这个我不清楚诶 y老师没加但是可以运行
    
    getline(cin, m);//可以用cin的getline读一整行 但是只能读string
    cout << m << endl;
    
    
    puts(s);//这个函数等价于下面这个printf 包括换行符
    printf("%s\n", s + 1);//从下标1开始输出 注意这里字符串用的是%s 不是%c
    // cout << s + 2 << endl;从下标2开始输出
    
    return 0;
}
