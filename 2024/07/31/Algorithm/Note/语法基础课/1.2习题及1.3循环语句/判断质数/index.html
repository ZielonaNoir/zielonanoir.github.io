#include <cstdio>

using namespace std;

int main(){
    int n;
    scanf("%d", &n);
    
    bool is_prime = true;

    for(int i = 2; i <= n - 1; i ++){ //这里可以不到 n - 1,到 n / 2就可以了
        if(n % i == 0){
            printf("%d\n", i);
            is_prime = false;
            break;
        }
    }
    
    if(is_prime) printf("是质数");
    else printf("不是质数");
    
    return 0;
}
