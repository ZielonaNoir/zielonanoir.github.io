# 数的范围 - 二分

#include <iostream>

using namespace std;

const int N = 1e6 + 10;

int n, m;
int q[N];

int main(){
    scanf("%d%d", &n, &m);
    for(int i = 0; i < n; i++){
        scanf("%d", &q[i]);
    }
    while(m--){
        int x;
        scanf("%d", &x);
        int l = 0, r = n - 1;
        while(l < r){
            int mid = l + r >> 1;
            if(q[mid] >= x) r = mid;
            else l = mid + 1;
        }
        if(q[l] != x) cout << "-1 -1" << endl;
        else{
            cout << l << ' ';
            int l = 0, r = n - 1;
            while(l < r){
                int mid = l + r + 1 >> 1;
                if(q[mid] <= x) l = mid;
                else r = mid - 1;
            }
            cout << l << endl;
        }
    }
    return 0;
}

#include <iostream>

using namespace std;

const int N = 1e6 + 10;

int n, m;
int q[N];

int main(){
    scanf("%d%d", &n, &m);
    for(int i = 0; i < n; i++){
        scanf("%d", &q[i]);
    }
    while(m--){
        int x;
        scanf("%d", &x);
        
        //二分
        int l = 0, r = n - 1;
        // l 为 0 r 为 x 的时候 是有问题的 因为 0.01 的二次方根是 0.1 不在 0.01 以下的区间内 所以要取 max(1, x)
        while(l < r){
            int mid = l + r >> 1;
            // >> 位运算 相当于 / 2 的 k 次方
            
            if(q[mid] >= x) r = mid;
            else l = mid + 1;
        }
        
        if(q[l] != x) cout << "-1 -1" << endl;
        else{
            cout << l << ' ';
            
            int l = 0, r = n - 1;
            while(l < r){
                int mid = l + r + 1 >> 1;
                //注意这里两个二分的区别
                if(q[mid] <= x) l = mid;
                //看是否是 l = mid 如果是的话前面就要补上 + 1;
                else r = mid - 1;
            }
            
            cout << l << endl;
            
        }
    }
    
    return 0;
 }

# 这里两个二分 一个是找左边一个是找右边

区别分别在 79 行 mid = l + r + 1 >> 1 和 83 行 r = mid - 1
